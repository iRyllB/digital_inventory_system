cmake_minimum_required(VERSION 3.19)
project(digital_inventory_system LANGUAGES CXX)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core Widgets)

qt_standard_project_setup()

# ðŸ”§ Add this to let AutoUic find the .ui files under ui/
set(CMAKE_AUTOUIC_SEARCH_PATHS ${CMAKE_CURRENT_SOURCE_DIR}/ui)

qt_add_executable(digital_inventory_system
    WIN32 MACOSX_BUNDLE

    main.cpp

    # Source files
    src/loginwindow.cpp
    src/mainwindow.cpp

    # Header files
    headers/loginwindow.h
    headers/mainwindow.h

    # UI files (CMake will still process these)
    ui/loginwindow.ui
    ui/mainwindow.ui
)

target_include_directories(digital_inventory_system
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/headers
)

target_link_libraries(digital_inventory_system
    PRIVATE
        Qt::Core
        Qt::Widgets
        Qt6::Gui
)

include(GNUInstallDirs)

install(TARGETS digital_inventory_system
    BUNDLE  DESTINATION .
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)

qt_generate_deploy_app_script(
    TARGET digital_inventory_system
    OUTPUT_SCRIPT deploy_script
    NO_UNSUPPORTED_PLATFORM_ERROR
)
install(SCRIPT ${deploy_script})
